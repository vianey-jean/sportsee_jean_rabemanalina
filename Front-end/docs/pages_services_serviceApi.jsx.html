import axios from 'axios'

/**
 * get USER_MAIN_DATA from API
 * @param {Number} id
 * @returns Object
 */


export async function getUserData({ id }) {
  try {
    const response = await axios.get(`http://localhost:3000/user/${id}`)
    return response.data
  } catch (error) {
    console.error(error)
  }
}
